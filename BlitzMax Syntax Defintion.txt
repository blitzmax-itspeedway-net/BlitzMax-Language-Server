BLITZMAX SYNTAX TREE

PROGRAM:    STRICT FRAMEWORK IMPORTS [BODY]

STRICT:     ["strict"|"superstrict" CRLF]

FRAMEWORK:  ["framework" string "." string CRLF]
    
IMPORTS:    ["import" string "." string CRLF ...]

BODY:       [DEFINTION ...]

DEFINTION:  [FUNCTION|TYPE|STRUCT|GLOBAL|LOCAL|CONST]

TYPE:       "type" string ["extends" string] CRLF [GLOBAL|FIELD|CONST|STRUCT|FUNCTION|METHOD] "end type"|"endtype" CRLF 
STRUCT:     "struct" string CRLF [GLOBAL|FIELD|CONST|STRUCT|FUNCTION|METHOD] "end struct"|"endstructe" CRLF 

FUNCTION:   "function" string [":" DATATYPE] "(" [ARGUMENT-LIST] ")" CRLF BLOCK "end function"|"endfunction" CRLF
METHOD:     "method" string [":" DATATYPE] "(" ARGUMENT-LIST ")" CRLF BLOCK "end method"|"endmethod" CRLF
    
GLOBAL:     "global" TYPEDEF
LOCAL:      "field" TYPEDEF
FIELD:      "local" TYPEDEF
CONST:      "const" TYPEDEF

DATATYPE:    ["%"|"!"|"$"|"int"|"float"|"double"|"string" etc...]

TYPEDEF:    string ":" DATATYPE ["=" EXPRESSION]

ARGUMENT-LIST:  
            [ TYPEDEF ["," TYPEDEF ...] ]

VALUE:      [ STRING | NUMBER ]

NUMBER:     Regex:([0-9]*(\.[0-9]*)?)
STRING:     Regex:([A-Za-z_][A-Za-z0-9_]*)
